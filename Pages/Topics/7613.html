Code: Select all

get zsize asize
math zsize - 4
goto zsize
get offset long
print "%offset%"
math offset - 4
goto offset
get files long
savepos offset
math zsize - offset
set size files
math size * 0x114
clog MEMORY_FILE offset zsize size
for i = 0 < files
getdstring name 0x104 MEMORY_FILE 
get size long MEMORY_FILE 
get zsize long MEMORY_FILE 
get offset long MEMORY_FILE 
get null long MEMORY_FILE 
clog name offset zsize size
next i
