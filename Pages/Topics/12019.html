The format seems straightforward enough...

Code: Select all

struct header
{
    char magic[4] = "TRES";
    int version;
    int entryCount;
    int unknown; // 16
};

struct entry
{
    int messageIndex;
    int keyTextOffset;
    int type;
    int valueTextOffset;
    int reserved1; // always 0?
    int reserved2; // always 0?
};

struct main
{
	header h;
	int entryPointers[h.entryCount]; // offset to entries
	entry entries[h.entryCount];
	// null terminated string array
};
