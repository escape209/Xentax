Are you sure that you use strings in Unicode?

https://github.com/spotify/sparkey/blob ... murHash3.c

C++

Code: Select all

unsigned int re7gethash(wchar_t * m_String)
{
	unsigned int dwLength;
	unsigned int dwHash;

	dwLength = -1i64;
	do
		++dwLength;
	while (*(WORD *)(m_String + dwLength));
	return MurmurHash3_x86_32(m_String, dwLength * 2, -1);
}
https://gist.github.com/automatonic/3725443

C#

Code: Select all

        public static uint r7gethash(string m_String)
        {
            uint dwHash = 0;
            byte[] pBuffer = Encoding.Unicode.GetBytes(m_String);

            using (MemoryStream TMemoryStream = new MemoryStream(pBuffer))
            {
                dwHash = MurMurHash3.Hash(TMemoryStream);
            }
            return dwHash;
        }
