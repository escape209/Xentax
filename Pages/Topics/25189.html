Code: Select all

findloc OFFSET STRING "M3D "
math i = 0
do
    goto OFFSET
    get DUMMY long
    findloc NEXT_OFFSET STRING "M3D " 0 ""

     if NEXT_OFFSET == ""

        get SIZE asize
    else
        math SIZE = NEXT_OFFSET

    endif
    math SIZE -= OFFSET
    string NAME p= OFFSET SIZE i
    STRING PATH = NAME

    log "" OFFSET SIZE
    math i += 1
    math OFFSET = NEXT_OFFSET

while NEXT_OFFSET != ""
