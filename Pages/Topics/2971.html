Of course:

Code: Select all

void decryptData(byte *data, uint32 size) {
    byte xorValues[3] = {0x42, 0x99, 0x80};
    for (uint32 ofs = 0; ofs < size; ofs++) {
        uint32 value = ofs % 3;
        data[ofs] ^= xorValues[value] - 0x6E;
    }
}
