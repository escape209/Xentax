Code: Select all

uint32_t hash_file_name(const char* input)
{
    if (input == NULL || *input == '\0')
    {
      return 0;
    }
    
    uint32_t hash = 0;
    for (const char* p = input; *p != '\0'; ++p)
    {
        hash *= 37;
        hash += tolower(*p);
    }
    return hash;
}
