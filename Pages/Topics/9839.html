DO
getdstring NAME 0x100
get OFFSET long
get SIZE long
log NAME OFFSET SIZE

Math OFFSET += SIZE

Math hh = MAX_OFFSET
Math hh -= OFFSET

GoTo OFFSET

While hh >= 0x100
##########
