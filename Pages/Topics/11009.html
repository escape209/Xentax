Here is a script, but it unpacks only a few dds and a lot of unknow files.

Code: Select all

goto 32
get files1 long
get files2 long
get files3 long
get files4 long
xmath files "files1+files2+files3+files4"
getdstring unk 8
for i = 0 < files
get size long
get id long
if id = 0x58534444
getdstring ddsdata 8
get size long
endif
savepos offset
log "" offset size
math offset += size
goto offset
next i
