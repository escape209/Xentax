Post by aluigi » Tue Dec 01, 2009 9:04 pm

ops my error, the following one should work (obviously I have used again a work-around but works at least for these 2 files):

Code: Select all

open FDSE "CDFILES.DAT" 0
#open FDSE "ARCHIVE.AR"  1

get DUMMY long
get BLAH long
goto 4
for i = 0 < BLAH
    get DUMMY long
    get DUMMY long
next i
get FILES long
print "XXX %FILES%"
get DUMMY long
get DUMMY long
get ENTRIES long
get DUMMY long
get DUMMY long
get DUMMY long
get DUMMY long
get DUMMY long
getdstring NAME 12
open FDSE NAME  1

math TMP = FILES
math TMP *= 4
savepos OFFSET
log MEMORY_FILE OFFSET TMP  # offsets
math OFFSET += TMP
log MEMORY_FILE2 OFFSET TMP # sizes

for i = 0 < FILES
    get OFFSET long MEMORY_FILE
    math OFFSET *= 8
    get SIZE long MEMORY_FILE2
    log "" OFFSET SIZE 1
next i
