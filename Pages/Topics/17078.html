this bms script will decompress your .gz samples

Code: Select all

# script for QuickBMS http://aluigi.altervista.org/quickbms.htm

comtype zlib_noerror
get NAME basename
do
	get ZSIZE long
	savepos OFFSET
	append
	clog NAME OFFSET ZSIZE ZSIZE
	append
	math OFFSET + ZSIZE
	goto OFFSET
while ZSIZE != 0
edit
first submesh in pc00_scl.elixir
