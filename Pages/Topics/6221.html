typedef unsigned char _BYTE;

void Decrypt(int a1, int a2, int a3)
{
  int v4;
  unsigned __int8 i;
  int v6; 
  char v7;

  v4 = 0;
  for ( i = 115; v4 < a2; *(_BYTE *)v6 = i ^ v7 )
  {
    v6 = v4 + a3;
    i = (i >> 7) + 2 * i;
    v7 = *(_BYTE *)(v4 + a3 + a1 - a3) ^ (v4 + v4 / 0xFFu);
    ++v4;
  }
}

a2 - size
a3 - destbuffer
