Here is a quick bms script to seperate the model parts

Code: Select all

get name2 basename
goto 0x4
get files long
math files / 16
goto 0x0
for i = 0 < files
getdstring type 4
get offset long
get size long
get unk01 long
set name name2
string name + .
string name + type
if size != 0
log name offset size
else
endif
next i
