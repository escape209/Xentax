Ore no Imouto ga Konna ni Kawaii wake ga Nai
My Little Sister Can't Be This Cute HappyEnd
BLJS-10238 / NPJB-00412
http://oreimo-game.channel.or.jp/

Quickbms script to extract the dat files.

Code: Select all

#Quickbms script by chrrox
#My Little Sister Can't Be This Cute HappyEnd (PS3)
comtype GZIP
idstring "GPDA64BY"
get TSIZE longlong
get NULL long
get FILES long
savepos TMP
for i = 0 < files
goto TMP
get OFFSET longlong
get NULL longlong
get SIZE longlong
get NOFF longlong
savepos TMP
goto NOFF
get NSIZE long
getdstring NAME NSIZE
set EXT EXTENSION NAME
if EXT == "dat"
log NAME OFFSET SIZE
else
clog NAME OFFSET SIZE SIZE
endif
next i
