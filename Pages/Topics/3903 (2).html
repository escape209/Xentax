Post by chrrox » Sat Jan 02, 2010 12:30 pm

if anyone wants a max 8 converter here is the script.

Code: Select all


/*
Special Thanks to everyone who helped with this.
Aluigi, Fatduck, Surveyor, and Asmodean
This is version .01
I will try to improve the script to add features as I get time.

To Do --
--Import Multiple objects at one time
--Auto Texture assigment
--Mesh Names
--Bones
--Weighting
--??
*/	

fname = GetOpenFileName caption:"Open Yukes Model File" types:"Yukes Model File(*.yonj)|*.yobj"
f = fopen fname "rb"   --open file in read only format

fn ReadBEword fstream = (
return (bit.swapBytes (readshort fstream #unsigned) 1 2)
)

fn ReadBElong fstream = (
long = readlong fstream
long = bit.swapBytes long 1 4
long = bit.swapBytes long 2 3
return long
)

fn ReadBEfloat fstream = (
return FloatSwap(readfloat fstream)
)


Vert_array = #()    --define arrays for verts, normals, UV and Faces
Normal_array = #()
UV_array = #()
Face_array = #()
IDstring = ReadBElong f --YOBJ
BoneOffset1 = ReadBElong f + 8 --Bone Data
Null1 = ReadBElong f --Null
BoneOffset2 = ReadBElong f + 8 --Bone Data Seems same as 1
Null2 = ReadBElong f --Null
Null3 = ReadBElong f --Null
MeshNumber = ReadBElong f --Null
Count2 = ReadBElong f --Null
Count3 = ReadBElong f --Null
Count4 = ReadBElong f --Null
BoneOffset3 = ReadBElong f --Null
BoneOffset4 = ReadBElong f --Null
BoneOffset5 = ReadBElong f --Null
Count5 = ReadBElong f --Null
Null4 = ReadBElong f --Null
Null5 = ReadBElong f --Null
Null6 = ReadBElong f --Null
Null7 = ReadBElong f --Null
--for aa = 1 to Count1 Do (
/*
Structure meshdata (
	VertexCount, FaceCount, Count1, Count2, Count3, Count4, Count5, Count6, Count7, Padding1, Count8, Count9, Count10, VertxOffset, WeightOffset, UVOffset, Count11, MaterialName, Count12, Count13, PartNum, PartOffset, DrawOffset, VertexNumber2, Count14, Count15, Count16, Count17, Count18
	)
	Table_Array = #() 
	for l = 1 to MeshNumber Do (
		
		
	)
	
*/	
for aa = 1 to MeshNumber Do (	
Vert_array = #()    --define arrays for verts, normals, UV and Faces
Normal_array = #()
UV_array = #()
Face_array = #()
	
	VertexCount = ReadBElong f --Null Vert count
    FaceCount = ReadBElong f --Null Face Count
	Count6 = ReadBElong f --Null
	Count7 = ReadBElong f --Null
	Count8 = ReadBElong f --Null
	Count9 = ReadBElong f --Null
	Count10 = ReadBElong f --Null
	fseek f 0x40 #seek_cur
	IDCount1 = ReadBElong f --Null
	IDCount2 = ReadBElong f --Null
	IDCount3 = ReadBElong f --Null
	VertexStartOffset = ReadBElong f + 12
	WeightOffset = ReadBElong f + 8 --Null
    UVOffset = ReadBElong f + 8 --Null
	Count11 = ReadBElong f --Null Num Textures?
	fseek f 0x10 #seek_cur --yBumpMap
	Count12 = ReadBElong f
	Count13 = ReadBElong f
	ParamNumber = ReadBElong f
	ParamOffset = ReadBElong f
	DrawOffset = ReadBElong f
	VertexNum2 = ReadBElong f
	Count14 = ReadBElong f
	fseek f 0x10 #seek_cur --Null Count
OffsetStart = ftell f
fseek f VertexStartOffset #seek_set
for i = 1 to VertexCount Do (
vx = ReadBEfloat f  --read xyz coordinates
vy = ReadBEfloat f
vz = ReadBEfloat f
append Vert_array [vx,vy,vz] --save verts to Vert_array
nx = ReadBEfloat f  --read xyz coordinates
ny = ReadBEfloat f
nz = ReadBEfloat f
unk4 = ReadBEfloat f
)

fseek f WeightOffset#seek_set
if IDCount1 == 1 Do (
for i = 1 to VertexCount Do (
unk5 = ReadBEfloat f  --read xyz coordinates
unk6 = ReadBEfloat f
)
)
if IDCount1 == 2 Do (
for i = 1 to VertexCount Do (
unk5 = ReadBEfloat f  --read xyz coordinates
unk6 = ReadBEfloat f
unk7 = ReadBEfloat f
unk8 = ReadBEfloat f
)
)
if IDCount1 == 3 Do (
for i = 1 to VertexCount Do (
unk5 = ReadBEfloat f  --read xyz coordinates
unk6 = ReadBEfloat f
unk7 = ReadBEfloat f
unk8 = ReadBEfloat f
unk9 = ReadBEfloat f  --read xyz coordinates
unk10 = ReadBEfloat f
)
)
if IDCount1 == 4 Do (
for i = 1 to VertexCount Do (
unk5 = ReadBEfloat f  --read xyz coordinates
unk6 = ReadBEfloat f
unk7 = ReadBEfloat f
unk8 = ReadBEfloat f
unk9 = ReadBEfloat f  --read xyz coordinates
unk10 = ReadBEfloat f
unk11 = ReadBEfloat f  --read xyz coordinates
unk12 = ReadBEfloat f
)
)
fseek f UVOffset#seek_set
for i = 1 to VertexCount Do (
tu = ReadBEfloat f    --read UV float value
tv = ReadBEfloat f * -1
append UV_array [tu,tv,0]  --save UVs to UV_array	
)
VarOff2 = ftell f
for i = 1 to FaceCount Do (
SIX = ReadBElong f --Null Always 6
FaceBytes = ReadBElong f --Null
FaceIndexOffset = ReadBElong f + 8
FaceIndexTableStart = ftell f
fseek f FaceIndexOffset#seek_set
FaceDirection = 1
f1 = ReadBEword f + 1   --read face indices, games are start form 0, but Max start from 1
f2 = ReadBEword f  + 1   --so we add 1 to each index
for i = 3 to FaceBytes do (    --already readf 2 verts, so start from 3	
f3 = ReadBEword f  + 1
FaceDirection *= -1	
if (f3 != f1) AND (f3 != f2) then (   --if not idential verts, create face
if FaceDirection > 0 then append Face_array [f1,f2,f3]     --base on face direction to create face
else append Face_array [f1,f3,f2]
)
f1 = f2     --shift the verts
f2 = f3
)
fseek f FaceIndexTableStart#seek_set
)

msh = mesh vertices:Vert_array faces:Face_array   --build mesh
msh.numTVerts = UV_array.count
buildTVFaces msh
for j = 1 to UV_array.count     do setTVert  msh j UV_array[j]
for j = 1 to Face_array.count   do setTVFace msh j Face_array[j]
for j = 1 to Normal_array.count do setNormal msh j Normal_array[j]

fseek f OffsetStart#seek_set	

)
fclose f     --close file
