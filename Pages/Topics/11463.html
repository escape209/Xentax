Simple script

Code: Select all

# Star Wars Knights of the Old Republic (iOS) (BZF format)
# 
# Written by Ekey (h4x0r)
# http://forum.xentax.com
# 
# script for QuickBMS http://quickbms.aluigi.org

comtype LZMA_DYNAMIC

idstring "BIFFV1\x20\x20"
get FILES long
get NULLS long
get DUMMY long
get PACKSIZE ASIZE
set j FILES
math j -= 1

for i = 0 < FILES
    get FILEID long
    get OFFSET long
    get SIZE long
    get DUMMY long
    savepos TEMP
	
	if i == j
	   set ZSIZE PACKSIZE
	   math ZSIZE - OFFSET
	   clog "" OFFSET ZSIZE SIZE
	else
	   get FILEID2 long
	   get OFFSET2 long
	   get SIZE2 long
	   get DUMMY2 long
	
	   math OFFSET2 -= OFFSET
	   set ZSIZE OFFSET2
	   clog "" OFFSET ZSIZE SIZE
	endif
	goto TEMP
next i
