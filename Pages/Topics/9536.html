Code: Select all

# FINAL FANTASY VII Remake (LGP format)
# 
# Written by Ekey (h4x0r)
# http://www.progamercity.net
# 
# script for QuickBMS http://quickbms.aluigi.org

get NULLS short
idstring "SQUARESOFT"
get FILES long

for i = 0 < FILES
	getdstring NAME 0x14
	get OFFSET long
	get DUMMY threebyte
	savepos TEMP
	math OFFSET += 0x14
	goto OFFSET
	get SIZE long
	math OFFSET += 0x4
	log NAME OFFSET SIZE
	goto TEMP
next i
