//Disney Infinity Research
// Cra0kalo/Mariokart64n/Luxox_18
//PC Little Endian


struct VBufHeader
{
// NOP
}

//ShaderDump VertexStructure

struct GPU_ASM
{
    vs_3_0
    def c0, 4, 1, 0, 1000
    def c1, 0.00100000005, 0, 0, 0
    dcl_blendindices v0
    dcl_blendweight v1
    dcl_normal v2
    dcl_position v3
    dcl_texcoord1 v4
    dcl_texcoord v5
    dcl_texcoord o0.xyz
    dcl_position o1
    dcl_texcoord1 o2
    dcl_texcoord2 o3.xy
    slt r0, v0, -v0
}

// Buffer
// Size 24
struct VBufStruc
{
float vd x
float vd y
float vd z
hfloat TextureU
hfloat TextureV
uint32 unknownPadding
uint32 VertexColor
}


// Buffer
// Stride 16
struct VBufStruc2
{
hfloat vn_x
hfloat vn_y
hfloat vn_z
hfloat vn_w

hfloat vt_x
hfloat vt_y
hfloat vt_z
hfloat vt_w


}



// Index Buffer
// Size 24
struct IndexBufStruc
{
uint16 face1
uint16 face2
uint16 face3
}
