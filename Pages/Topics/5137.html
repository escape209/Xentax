The UTF content is encrypted. This code can decrypt it. But decompress failed ...

Code: Select all

int decrypt_cpk(u8 *buf, int length)
{
	u32 m, t, d;
	int i;

	m = 0x0000655f;
	t = 0x00004115;

	for(i=0; i<length; i++){
		d = buf[i];
		d = d^(m&0xff);
		buf[i] = d;
		m = m*t;
	}

	return 0;
}
