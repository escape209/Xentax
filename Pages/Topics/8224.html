Post by chrrox » Wed Feb 08, 2012 7:10 pm

the format is a little different.
can you post the full game client download link.

Code: Select all

goto 0x104
get files short
get null short
for i = 0 < files
get id long
putarray 0 i id
next i
for i = 0 < files
get id1 long
putarray 1 i id1
next i
for i = 0 < files
getarray OFFSET 1 i
if OFFSET == -1
getdstring null 0x310 #this is the recursive directory info but we can skip it in this game
endif
if OFFSET != -1
getdstring basename 0xF8
get SIZE long
get UNK01 long
get UNK02 long
getdstring NULL 0x2
getdstring NAME 0x20A
putarray 2 i SIZE
putarray 3 i NAME
endif
next i
savepos base
for i = 0 < files
getarray OFFSET 1 i
getarray SIZE 2 i
getarray NAME 3 i
math OFFSET + base
if NAME != 0
log NAME OFFSET SIZE
endif
next i
