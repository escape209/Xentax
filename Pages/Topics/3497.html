That's what I have so far, don't have the time to continue.

Code: Select all

string name;
uint uk;
uint uk2;
uint uk3;
int uk4;
uint uk5;

struct Markers
{
    char ID[8];
    uint numGroups;
    uint size;
    uint uk;
    int uk4;
    uint uk5;
    uint uk6;
    uint uk7;
    char uk8[uk7];
    uint uk9;
    string uk10;
};

struct MeshGroup
{
    char ID[8];
    uint numGroups;
    uint size;
    uint uk;
    int uk4;
    uint uk5;
    Markers markers;    
};

struct Mesh
{
    char ID[8];
    uint numGroups;
    uint size;
    uint uk;
    int uk4;
    uint uk5;
    MeshGroup groups[numGroups];
};

struct Model
{
    char ID[8];
    uint numMeshes; // ?
    uint size;
    uint uk;
    int uk4;
    uint uk5;
    string name;
    Mesh meshes[numMeshes];
} model;
