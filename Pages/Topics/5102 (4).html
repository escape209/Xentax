Post by Ekey » Thu Feb 27, 2014 4:10 pm

Done.

Code: Select all

# Castlevania: Lords of Shadow 2 (PC) (PACKED format)
# 
# Written by Ekey (h4x0r)
# http://forum.xentax.com
# 
# script for QuickBMS http://quickbms.aluigi.org

comtype unzip_dynamic

idstring "BFPK"
get VERSION short
get FLAG short
get FILES long

for i = 0 < FILES
    get NSIZE long
    getdstring NAME NSIZE
    get SIZE long
    get OFFSET longlong
    savepos TEMP
	
	if FLAG = 0
        log NAME OFFSET SIZE
	elseif FLAG = 1
        goto OFFSET
        get ZSIZE long
        savepos OFFSET
        clog NAME OFFSET ZSIZE SIZE
    endif
    goto TEMP
next i
