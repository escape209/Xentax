//--------------------------------------
//--- 010 Editor v5.0 Binary Template

// File:     menu.mgb
// Author:   wrs
// Revision: 1
// Purpose:  mgb parser
//--------------------------------------

LittleEndian();

char hdr[4]; // "MAGM"
uint unknown_52545;
uint unknown_4011;
ushort unknown;

ushort width;  // 640
ushort height; // 480

FSeek(250); // ??
uint num_1;
FSeek(256+7);
uint num;

Assert(num==num_1);

struct {
  uint unknown;
  uint strlen;
  wchar_t str[strlen];

  Printf("%08x : %s\n", unknown, str);

} strs[num] <optimize=false>;
