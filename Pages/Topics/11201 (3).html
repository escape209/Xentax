I'm using this for rotation:

Code: Select all

	for x = 1 to num_Rotation_Animation do
		(
			Tick_Time = ((readlong f #unsigned) / 80) --Frame
		  
			vx = readfloat f
			vy = readfloat f
			vz = readfloat f
			vw = readfloat f
			
			Arot = quat vx vy vz vw
			
			if ANode != undefined then
			(
				with animate on 
				(
				   at time Tick_Time
				   (

			)
		)
