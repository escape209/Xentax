Section 'G1MS' - bones
endianess = big

offset - position of G1MS section
string[8] - "G1MS0032"
dword - unknow
dword - seek
word - unknow
dword - nBones
go to (offset+seek)

nBones x (48bytes){
float[3] - bone local scale
dword - bone idParent
float[4] - bone local rotation(qx,qy,qz,qw)
float[4] - bone local position
}
