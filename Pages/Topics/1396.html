uint32 {4} - Version (1)
uint32 {4} - Number Of Files

uint32 {4} - File Offset
uint32 {4} - File Size

// FILENAME DIRECTORY

uint32 {4} - Filename Length (including null)
char {X} - Filename
byte {1} - null Filename Terminator
