            vector = cksum2_magic(idx) Xor (vector << 8)

Script with new formatting hash ID

Code: Select all

open FDDE bix 1
goto 0x58 1
get numFiles long 1
goto 0x90 1

for i = 0 < numFiles
    get hash long 1
    get Offset long 1
    get unk1 long 1
    get CSize long 1
    get unk2 long 1
    get USize long 1
   
    set SizeEmpty unk1
    math SizeEmpty &= 0xFFF
    math Offset *= 4
    math Offset += SizeEmpty

    string NAME p= "%08X" hash
   
    if CSize == 0
      log NAME Offset USize
    else
      log NAME Offset CSize
    endif
next i
