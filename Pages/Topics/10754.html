PAK does not contain file names, only hash like MD5

Code: Select all

# Castle of Illusion (PAK format) 0.1
# 
# Written by Ekey (h4x0r)
# http://forum.xentax.com
# 
# script for QuickBMS http://quickbms.aluigi.org

comtype lz4

goto 0x8
get FILES long
get SUBFILES long
goto 0xaa0a

for i = 0 < FILES
	getdstring NAME 32
	get ZSIZE long
	get SIZE long
	get OFFSET long
	get UNKNOWN1 long
	get UNKNOWN2 long
	get UNKNOWN3 long
	
	if ZSIZE == SIZE
		log NAME OFFSET SIZE
	else
		clog NAME OFFSET ZSIZE SIZE
	endif
next i
