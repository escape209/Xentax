Use this to unpack the resource:

And Please Upload The Clinet!!!

Code: Select all

# Game: Busou Shinki [PC]
# http://www.busou.konami.jp/shinki_net/
# by Fatduck    Apr2012
# script for QuickBMS http://aluigi.org/papers.htm#quickbms

idstring "PKDT"
get ver long
get BINSIZE long
get OFSRESBASE long
get NUMRES long
get OFSTBL long
for i = 0 < NUMRES
   goto OFSTBL
   get IDXSIZE long
   math OFSTBL += IDXSIZE
   get OFSRES long
   math OFSRES += 0x10
   get RESSIZE long
   get UKN long
   get RESNAME string
   log RESNAME OFSRES RESSIZE
next i
