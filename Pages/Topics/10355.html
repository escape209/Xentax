#Eador Masters of the Broken World
#quickbms script by chrrox
#extract .pac file
idstring "PACK"
get VER long
get TBLOFF longlong
get FILES longlong
goto TBLOFF
for i = 0 < FILES
filexor "0x98 0x83 0x84 0x8E 0x82 0x9F 0x83 0x92 0x9E 0x83 0x82 0x9A 0x8F 0x84 0x9F 0x89" TBLOFF
get OFFSET longlong
get SIZE longlong
get NSIZE long
get UNK long
getdstring NAME NSIZE
print "%NAME%"
