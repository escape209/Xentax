Post by Ekey » Sun Mar 23, 2014 9:14 pm

Well here script for unpack

Code: Select all

# Q-Dragonball (iOS) (PAK format)
# 
# Written by Ekey (h4x0r)
# http://forum.xentax.com
# 
# script for QuickBMS http://quickbms.aluigi.org

idstring "PACK"
get MINOR_VERSION short
get MAJOR_VERSION short
get FLAG long
get UNKNOWN short
get FILES short

for i = 0 < FILES
    do
      get HASH1 long
      get HASH2 long
      get HASH3 long
      get DATAOFFSET long
      savepos TEMP
    while DATAOFFSET <= 0xFFFFFFFF
	
    goto DATAOFFSET
	
    get NSIZE short
    getdstring NAME NSIZE
    get COMPFLAG long
    get SIZE long
    get ZSIZE long
    savepos OFFSET
	
    if SIZE == ZSIZE
        log NAME OFFSET SIZE
    else
        clog NAME OFFSET ZSIZE SIZE
    endif
	goto TEMP
next i
