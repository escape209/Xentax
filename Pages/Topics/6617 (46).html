				--	Add Weights
				
				arxImportUI.pgbProgress.value = 0
				
				for x = 1 to srtArxMeshData[i].vertCount do (
				
					mySkin = newMesh.skin
					
					for y = 1 to 8 do (
					
						if srtArxMeshData[i].arxBones[x][y] > 0 then (
