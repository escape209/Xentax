ADDON1.dem

Code: Select all

Big Endian
IDSTRING       - 0x14 -       QUANTICDREAMTABINDEX
Version          - 0x4 -         00 00 00 11
Section Count -0x4 -          00 00 00 07 
for each section count read the sub section header
example
4 longs
00 00 0F A0 00 00 00 01 00 00 00 EA 00 00 00 03
Then we have that last long that means the sub section contains 3 files?
each of these has this data

Post by Rick » Tue Oct 22, 2013 8:52 pm

.idm = file table (this is the file encrypted with the SDAT container)

Code: Select all

struct Header
{
    char Magic[20];
    uint32 Unknown00<format=hex>;
    uint32 TableCount;
    uint32 Unknown08<format=hex>;
};

struct Resource
{
    uint32 ResourceId;
    uint32 DataOffset;
    uint32 DataSize;
    uint32 Unknown0C;
    ubyte ContainerIndex;
    ubyte __Padding[3];
};

struct Table
{
    uint32 TypeId;
    uint32 ResourceCount;
    uint32 ResourceInfoOffset;

    if (ResourceCount > 0)
    {
        local int64 previousOffset = FTell();
        FSeek(20 + ResourceInfoOffset);
        uint32 FastLookupTable[256];
        Resource Resources[ResourceCount];
        FSeek(previousOffset);
    }
};

struct TabIdMem
{
    Header header;
    Table tables[header.TableCount]<optimize=false>;
};

BigEndian();
TabIdMem root;
.dem = maps hashed names to resource IDs. Not all resources have names.

Post by chrrox » Thu Oct 24, 2013 2:47 am

Here is a crude extractor for now if people want to start looking at files.
You should decrypt the sdat file with
TrueAncestor EDAT Rebuilder
by JjKkYu
Verision 1.42
then just rename the extension of the sdat to anything and place it with the original file.

Code: Select all

endian big
idstring "QUANTICDREAMTABIDMEM"
get NULL long
get DIR long
get NULL long
savepos TBL
for i = 0 < DIR
goto TBL
get TYPE long
get FILES long
get TBLOFF long
math TBLOFF + 0x414
savepos TBL
goto TBLOFF
for a = 0 < FILES
get FID long
get OFFSET long
get SIZE long
get UNK long
get ARC byte
get NULL threebyte
get NAME basename
if arc < 1
string NAME + .dat
elif arc < 10
string NAME + .d0
string NAME + ARC
else
string NAME + .d
string NAME + ARC
endif
#print "%NAME%"
open FDSE NAME 1
log "" OFFSET SIZE 1
next a
next i

Post by Ekey » Mon Oct 28, 2013 9:29 pm

Well here simple script for decompress segs

Code: Select all

# Beyond: Two Souls (SEGS format) 0.2a
#
# Based on script AR segs archives by Aluigi
# Modified by Ekey (h4x0r)
#
# http://forum.xentax.com
#
# script for QuickBMS http://quickbms.aluigi.org

comtype deflate

get NAME basename
get EXT extension
string NAME += "_unpacked."
string NAME += EXT

endian big
idstring "segs"
get VERSION short
get CHUNKS short
get FULL_SIZE long
get FULL_ZSIZE long

set TABLESIZE = CHUNKS
math TABLESIZE *= 8
math TABLESIZE += 16

log MEMORY_FILE 0 0
append

for i = 0 < CHUNKS
    get ZSIZE short
    get SIZE short
    get OFFSET long
    math OFFSET -= 1
    math OFFSET += TABLESIZE

    if CHUNKS < 100
        math OFFSET += 8
    endif

    if SIZE == 0
        math SIZE = 0x00010000
    endif
	
    clog MEMORY_FILE OFFSET ZSIZE SIZE
    next i
  append
  log NAME 0 FULL_SIZE MEMORY_FILE
endif
Here example cutted part from 000000000000006e.seg > offset 0xDE , size 524426
